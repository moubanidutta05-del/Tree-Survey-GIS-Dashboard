<script>

var map = L.map('map').setView([20.59,78.96],5);

var satellite = L.tileLayer(
'https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}'
);

var labels = L.tileLayer(
'https://services.arcgisonline.com/ArcGIS/rest/services/Reference/World_Boundaries_and_Places/MapServer/tile/{z}/{y}/{x}'
);

var hybrid = L.layerGroup([satellite,labels]).addTo(map);


var trees = [
{"TreeName":"Neem","TreeID":1,"Height":"10m","Health":"Good","lat":22.57,"lng":88.36},
{"TreeName":"Neem","TreeID":2,"Height":"9m","Health":"Good","lat":22.58,"lng":88.35},
{"TreeName":"Mango","TreeID":3,"Height":"12m","Health":"Moderate","lat":22.56,"lng":88.37},
{"TreeName":"Peepal","TreeID":4,"Height":"14m","Health":"Good","lat":22.55,"lng":88.38},
{"TreeName":"Mango","TreeID":5,"Height":"11m","Health":"Poor","lat":22.54,"lng":88.36}
];

trees.forEach(function(t){

var marker = L.marker([t.lat,t.lng]).addTo(map);

marker.bindPopup(
"<b>Tree Name:</b> "+t.TreeName+
"<br><b>Tree ID:</b> "+t.TreeID+
"<br><b>Height:</b> "+t.Height+
"<br><b>Health:</b> "+t.Health
);

});


L.control.scale().addTo(map);


var legend = L.control({position:'bottomright'});

legend.onAdd = function(map){

var div = L.DomUtil.create('div','legend');

div.innerHTML =
"<b>Legend</b><br>"+
"📍 Tree Location";

return div;

};

legend.addTo(map);



var freq = {};

trees.forEach(function(t){

freq[t.TreeName] = (freq[t.TreeName] || 0) + 1;

});

var labels = Object.keys(freq);

var values = Object.values(freq);


var ctx = document.getElementById('pieChart');

new Chart(ctx,{
type:'pie',
data:{
labels:labels,
datasets:[{
data:values
}]
},
options:{
responsive:true
}
});

</script>
